
## H-bridge (outputs)
- PE1 -> ENA
- PE2 -> IN1A
- PE3 -> IN2A
- PA5 -> ENB
- PA6 -> IN1B
- PA7 -> IN2B
## Protection board (inputs)
- PA2 <- Index0 (Pan) (interrupt)
- PA3 <- Index1 (Tilt) (interrupt)
- PC4 <- Sensor1A (interrupt)
- PC5 <- Sensor1B
- PC6 <- Sensor2A (interrupt)
- PC7 <- Sensor2B
## SPI 
- PD0 -> SSI1Clk
- PD1 -> SSI1Fss
- PD2 <- SSI1Rx
- PD3 -> SSI1Tx
